数据包格式为 4字节包长度 + 包体

包体格式定义：
2字节 客户端信令 + 包数据

包数据格式定义：

COMMAND_REG：
   AppID uint16
   AppType uint16
   AppName的长度 uint32
   AppName char[]  // 20字节以内

COMMAND_REG_RESPONSE：
   RegResult uint8 1=成功。0=失败

COMMAND_KEEP_ALIVE：

COMMAND_KEEP_ALIVE_RESPONSE：

COMMAND_QUERY_APP：
   Serverid uint16

COMMAND_QUERY_APP_RESPONSE：
   Serverid uint16
   AppCount uint16 //App数量
   ...
   Appid uint16
   isadd uint8   // 0,删除App；1,增加App
   ...

COMMAND_SYN_APP：
   Serverid uint16
   AppCount uint16 //App数量
   ...
   Appid uint16
   isadd uint8   // 0,删除App；1,增加App
   ...

COMMAND_QUERY_SERVER：

COMMAND_QUERY_SERVER_RESPONSE：
   ServerCount uint16 //Server数量
   ...
   Serverid uint16
   Stationid char[33]
   online uint8
   ...

COMMAND_PACKET    0x3000
   SourServerid uint16
   DestServerid uint16
   SourAppid uint16
   DestAppid uint16
   connid uint32  // 备用
   dataid uint32   //未分片数据包编号，自增
   datalength uint32 //数据总长度
   offset uint32 //数据偏移量
   
   /////////////////////////////
   数据内部格式
   md5 char[16]
   data char[]

COMMAND_PACKET_RECEIVED    0x3001
   SourServerid uint16
   DestServerid uint16
   SourAppid uint16
   DestAppid uint16
   connid uint32  // 备用
   dataid uint32  //数据包编号

COMMAND_PACKET_SERVERDOWN    0x3002
   Serverid uint16
   dataid uint32  //数据包编号
   
COMMAND_PACKET_APPDOWN    0x3003
   Serverid uint16
   SourAppid uint16
   DestAppid uint16
   dataid uint32  //数据包编号
   
COMMAND_PACKET_REJECT    0x3004
   SourServerid uint16
   DestServerid uint16
   SourAppid uint16
   DestAppid uint16
   connid uint32  // 备用
   dataid uint32  //数据包编号
   
COMMAND_PACKET_ERR    0x3005
   SourServerid uint16
   DestServerid uint16
   SourAppid uint16
   DestAppid uint16
   connid uint32  // 备用
   dataid uint32  //数据包编号
